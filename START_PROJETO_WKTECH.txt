CREATE SCHEMA 'wktech';

CREATE TABLE `wktech`.`cliente` (
  `CODIGO` INT NOT NULL AUTO_INCREMENT,
  `NOME` VARCHAR(60) NOT NULL,
  `Cidade` VARCHAR(60) NOT NULL,
  `Uf` VARCHAR(2) NULL,
  PRIMARY KEY (`CODIGO`))
COMMENT = 'Tabela de Cadastro de Clientes.';

CREATE TABLE `wktech`.`produto` (
  `CODIGO` INT NOT NULL AUTO_INCREMENT,
  `DESCRICAO` VARCHAR(45) NOT NULL,
  `PRECO` DOUBLE NOT NULL,
  PRIMARY KEY (`CODIGO`));
  
CREATE TABLE `wktech`.`pedido` (
  `IDPEDIDO` INT NOT NULL AUTO_INCREMENT,
  `NUMERO` INT UNSIGNED NOT NULL,
  `CODCLI` INT NOT NULL,
  `DTCADASTRO` DATE NULL,
  `TOTAL` DOUBLE NULL,
  PRIMARY KEY (`IDPEDIDO`),
  INDEX `FKCLIENTE_idx` (`CODCLI` ASC) VISIBLE,
  CONSTRAINT `FKCLIENTE`
    FOREIGN KEY (`CODCLI`)
    REFERENCES `wktech`.`cliente` (`CODIGO`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
COMMENT = 'Tabela de Cadastro de Pedidos';

CREATE TABLE `wktech`.`itenspedido` (
  `IDITEM` INT NOT NULL AUTO_INCREMENT,
  `IDPEDIDO` INT NOT NULL,
  `CODPROD` INT NOT NULL,
  `QTD` DOUBLE NOT NULL,
  `PRECO` DOUBLE NOT NULL,
  `TOTALITEM` DOUBLE NULL,
  PRIMARY KEY (`IDITEM`),
  INDEX `FK_PEDIDO_idx` (`IDPEDIDO` ASC) VISIBLE,
  INDEX `FK_PRODUTO_idx` (`CODPROD` ASC) VISIBLE,
  CONSTRAINT `FK_PEDIDO`
    FOREIGN KEY (`IDPEDIDO`)
    REFERENCES `wktech`.`pedido` (`IDPEDIDO`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `FK_PRODUTO`
    FOREIGN KEY (`CODPROD`)
    REFERENCES `wktech`.`produto` (`CODIGO`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
COMMENT = 'Tabela de Itens do Pedido';
